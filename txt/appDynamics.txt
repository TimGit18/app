public static String getHostName() {
    String hostName = "";
    // Den Rechnernamen aus der Umgebung holen
    try {
        hostName = InetAddress.getLocalHost().getHostName();
    } catch (IOException ioException) {
        System.err.println(ioException.getMessage());
    }
    return hostName;
}
